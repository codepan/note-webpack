4.5 多页应用打包

// webpack.config.js
const path = require('path')
const HtmlWebpackPlugin = require('html-webpack-plugin')

module.exports = {
  // mode: 'production',
  mode: 'development',
  entry: {
    index: './src/index.js',
    list: './src/list.js'
  },
  output: {
    filename: '[name].js',
    path: path.resolve(__dirname, 'dist')
  },
  module: {
    rules: [
      {
        test: /\.js$/,
        exclude: /node_modules/,
        use: ['babel-loader']
      }
    ]
  },
  plugins: [
    new HtmlWebpackPlugin({
      filename: 'index.html',
      template: 'index.html',
      chunks: ['index']
    }),
    new HtmlWebpackPlugin({
      filename: 'list.html',
      template: 'index.html',
      chunks: ['list']
    })
  ]
}

我们看到如果有很多html页面，我们就需要new很多次html-webpack-plugin，所以我们可以将其通过编码的方式，自动生成

// webpack.config.js
const path = require('path')
const HtmlWebpackPlugin = require('html-webpack-plugin')
const { CleanWebpackPlugin } = require('clean-webpack-plugin')

let config = {
  mode: 'development',
  entry: {
    index: './src/index.js',
    list: './src/list.js',
    me: './src/me.js'
  },
  output: {
    filename: '[name].js',
    path: path.resolve(__dirname, 'dist')
  },
  module: {
    rules: [
      {
        test: /\.js$/,
        exclude: /node_modules/,
        use: ['babel-loader']
      }
    ]
  }
}

const generatePlugins = config => {
  let plugins = [
    new CleanWebpackPlugin()
  ]
  Object.keys(config.entry).forEach(entry => {
    plugins.push(new HtmlWebpackPlugin({
      template: 'index.html',
      filename: `${entry}.html`,
      chunks: [`${entry}`]
    }))
  })

  return plugins
}

config.plugins = generatePlugins(config)

module.exports = config

可以看出，多页面应用打包没有什么神奇和难懂的地方，总结起来就三点：

  • entry为对象，每个key代表打包产出的js文件名，value代表入口js文件
  • new多个html-webpack-plugin
  • html-webpack-plugin 的chunks参数指定将来产出的html文件引入哪个js文件，值为数组，数组元素为entry对象的key
